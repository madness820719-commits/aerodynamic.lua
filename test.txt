function getAny()
	for _,pl in ipairs(game:GetService("Players"):GetPlayers()) do
		if pl~=lp and pl.Character then
			h=pl.Character:FindFirstChild("HumanoidRootPart")
			hum=pl.Character:FindFirstChildOfClass("Humanoid")
			if h and hum and hum.Health>0 then return pl.Character,pl end
		end
	end
end

function getNearest()
	best,bd,bp=nil,math.huge,nil
	me=lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
	if not me then return nil,nil end
	for _,pl in ipairs(game:GetService("Players"):GetPlayers()) do
		if pl~=lp and pl.Character then
			h=pl.Character:FindFirstChild("HumanoidRootPart")
			hum=pl.Character:FindFirstChildOfClass("Humanoid")
			if h and hum and hum.Health>0 then
				d=(h.Position-me.Position).Magnitude
				if d<bd then best=pl.Character bd=d bp=pl end
			end
		end
	end
	return best,bp
end

function mkSound()
	pcall(function()
		if hitSnd and hitSnd.Parent then hitSnd:Destroy() end
		parent=workspace.CurrentCamera or game:GetService("SoundService")
		hitSnd=Instance.new("Sound")
		hitSnd.Volume=hitSndVol hitSnd.PlaybackSpeed=hitSndPitch
		hitSnd.RollOffMaxDistance=999999 hitSnd.RollOffMode=Enum.RollOffMode.InverseTapered
		hitSnd.SoundId="rbxassetid://"..(customHitSndId~="" and customHitSndId or soundIds[sndIdx])
		hitSnd.Parent=parent
	end)
end

function startRF()
	if rfConn then rfConn:Disconnect() end
	t=0 swingT=0
	rfConn=rs.Heartbeat:Connect(function(dt)
		if not S.rf then return end
		t=t+dt swingT=swingT+dt
		chr2=lp.Character if not chr2 then return end
		if t>=fireRate then
			t=0
			for _,tool in ipairs(chr2:GetChildren()) do
				if tool:IsA("Tool") then
					pcall(function()tool:Activate()end)
					if rfDouble then task.delay(0.002,function()pcall(function()tool:Activate()end)end) end
					if S.rfMultiActivate then
						for i=1,rfMultiCount do task.delay(0.001*i,function()pcall(function()tool:Activate()end)end) end
					end
					if S.rfBypassMode or rfBypass then
						pcall(function()
							for _,c in ipairs(getconnections(tool.Activated)) do pcall(c.Fire,c) end
						end)
					end
				end
			end
		end
		if S.rfAutoSwing and swingT>=rfSwingRate then
			swingT=0
			for _,tool in ipairs(chr2:GetChildren()) do
				if tool:IsA("Tool") then
					pcall(function()
						handle=tool:FindFirstChild("Handle")
						if handle then
							weld=handle:FindFirstChildOfClass("Motor6D") or handle:FindFirstChildOfClass("Weld")
							if weld then weld.C0=weld.C0*CFrame.Angles(0,math.rad(30),0) end
						end
					end)
				end
			end
		end
	end)
end
function stopRF() if rfConn then rfConn:Disconnect() rfConn=nil end end

function doRivalsBypass(target)
	if not S.rivalsBypass then return end
	pcall(function()
		chr=lp.Character if not chr then return end
		myhrp=chr:FindFirstChild("HumanoidRootPart") if not myhrp then return end
		tchar=target if not tchar then tchar=getAny() end if not tchar then return end
		thrp=tchar:FindFirstChild("HumanoidRootPart") if not thrp then return end
		thum=tchar:FindFirstChildOfClass("Humanoid") if not thum or thum.Health<=0 then return end
		if rivalsHitRemote and rivalsBpMode=="remote" then
			if rivalsHitRemote:IsA("RemoteEvent") then
				pcall(function()rivalsHitRemote:FireServer(thrp,rivalsDmg)end)
				pcall(function()rivalsHitRemote:FireServer(tchar,rivalsDmg)end)
			elseif rivalsHitRemote:IsA("RemoteFunction") then
				pcall(function()rivalsHitRemote:InvokeServer(thrp,rivalsDmg)end)
			end
		end
		if rivalsBpMode=="touch" or rivalsBpMode=="both" then
			for _,tool in ipairs(chr:GetChildren()) do
				if tool:IsA("Tool") then
					handle=tool:FindFirstChild("Handle")
					if handle then
						for _,p in ipairs(tchar:GetDescendants()) do
							if p:IsA("BasePart") then
								pcall(firetouchinterest,handle,p,0) pcall(firetouchinterest,handle,p,1)
							end
						end
					end
				end
			end
		end
		if rivalsBpMode=="all" or rivalsBpMode=="both" then
			for _,v in ipairs(game:GetService("ReplicatedStorage"):GetDescendants()) do
				if v:IsA("RemoteEvent") then
					pcall(function()v:FireServer(thrp)end) pcall(function()v:FireServer(tchar)end)
				end
			end
		end
		pcall(function()
			for _,p in ipairs(tchar:GetDescendants()) do
				if p:IsA("ProximityPrompt") then fireproximityprompt(p) end
			end
		end)
	end)
end

function updateSA()
	if saConn then saConn:Disconnect() saConn=nil end
	if not S.sa then return end
	saConn=rs.RenderStepped:Connect(function()
		if not S.sa then return end
		pcall(function()
			e=getAny() if not e then return end
			torso=e:FindFirstChild("UpperTorso") or e:FindFirstChild("Torso") or e:FindFirstChild("HumanoidRootPart")
			if not torso then return end
			cam=workspace.CurrentCamera if not cam then return end
			vel=Vector3.zero
			pcall(function()
				h=e:FindFirstChild("HumanoidRootPart")
				if h then vel=h.AssemblyLinearVelocity end
			end)
			target=torso.Position+vel*saPred
			dir=(target-cam.CFrame.Position).Unit
			cam.CFrame=CFrame.new(cam.CFrame.Position,cam.CFrame.Position+dir)
		end)
	end)
end

-- helper: apply modifiers to xyz for void/orbit
function applyVSModifiers(x,y,z)
	if vsNoiseX~=0 then x=x+(math.random()-0.5)*vsNoiseX*2 end
	if vsNoiseY~=0 then y=y+(math.random()-0.5)*vsNoiseY*2 end
	if vsNoiseZ~=0 then z=z+(math.random()-0.5)*vsNoiseZ*2 end
	x=x*vsScaleX+vsOffsetX
	y=y*vsScaleY+vsOffsetY
	z=z*vsScaleZ+vsOffsetZ
	if vsPulseAmt~=0 then
		vsPulseT=(vsPulseT+0.1)%(math.pi*2)
		local pv=math.sin(vsPulseT)*vsPulseAmt
		x=x+pv y=y+pv z=z+pv
	end
	if vsWaveAmt~=0 then
		vsWaveT=(vsWaveT+0.08)%(math.pi*2)
		x=x+math.sin(vsWaveT)*vsWaveAmt
		z=z+math.cos(vsWaveT)*vsWaveAmt
	end
	if vsInvertX then x=-x end
	if vsInvertZ then z=-z end
	if vsSwapXZ then x,z=z,x end
	if vsAbsCoords then x=math.abs(x) y=math.abs(y) z=math.abs(z) end
	if vsLockY then y=vsLockYVal end
	return x,y,z
end

	if num("rfMultiCount") then rfMultiCount=d.rfMultiCount end if num("rfSwingRate") then rfSwingRate=d.rfSwingRate end

	end
	tmap={tVoid="voidSpam",tOrbit="orbitT",tTpa="tpAura",tBypass="bypassT",tRiot="riotAbuseT",tRf="rfT",tSa="saT",tAa="aaT",tMotion="motionT",tCh="chT",tEspBox="espBoxT",tEspChams="espChamsT",tEspName="espNameT",tEspHealth="espHealthT",tEspDist="espDistT",tEspTracer="espTracerT",tEspSkeleton="espSkeletonT",tEspCorner="espCornerBoxT",tHitNotify="hitNotifyT",tKillNotify="killNotifyT",tHitSound="hitSndT",tKillSound="killSndT",tStudsUI="studsUITogT",tRivals="rivalsT"}

-- core tp function for void spam
function computeVSCoords(hrp)
	x,y,z=0,0,0
	pat=vsPattern
	if pat=="custom" then
		x=vsCustomX y=vsCustomY z=vsCustomZ
		return x,y,z
	elseif pat=="static" then
		x=vsStaticX y=vsStaticY+vsBoostY z=vsStaticZ
	elseif pat=="void" then
		x=(math.random()-0.5)*2*vsXDist
		y=vsYCurrent+vsBoostY
		z=(math.random()-0.5)*2*vsZDist
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="everywhere" then
		x=(math.random()-0.5)*2*vsXDist
		y=(math.random()-0.5)*2*vsYDist
		z=(math.random()-0.5)*2*vsZDist
	elseif pat=="random" then
		x=(math.random()-0.5)*2*vsRadius
		y=vsYCurrent+vsBoostY+(math.random()-0.5)*vsRadius*0.1
		z=(math.random()-0.5)*2*vsRadius
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="circular" then
		vsAngle=(vsAngle+0.15)%(math.pi*2)
		x=math.cos(vsAngle)*vsRadius
		y=vsYCurrent+vsBoostY
		z=math.sin(vsAngle)*vsRadius
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="spiral" then
		vsAngle=(vsAngle+0.12)%(math.pi*2)
		r=vsRadius*(0.5+0.5*math.abs(math.sin(vsAngle*0.3)))
		x=math.cos(vsAngle)*r
		y=vsYCurrent+vsBoostY+vsAngle*500
		z=math.sin(vsAngle)*r
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="figure8" then
		vsAngle=(vsAngle+0.1)%(math.pi*2)
		x=vsRadius*math.sin(vsAngle)
		y=vsYCurrent+vsBoostY
		z=vsRadius*math.sin(vsAngle)*math.cos(vsAngle)
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="zigzag" then
		vsAngle=(vsAngle+0.12)%(math.pi*2)
		x=vsXDist*math.sin(vsAngle)
		y=vsYCurrent+vsBoostY
		z=vsZDist*math.cos(vsAngle*2)
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="sine" then
		vsAngle=(vsAngle+0.1)%(math.pi*2)
		x=(math.random()-0.5)*2*vsXDist
		y=vsYCurrent+vsBoostY+vsRadius*math.sin(vsAngle)
		z=(math.random()-0.5)*2*vsZDist
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="grid" then
		gx=math.floor(math.random()*8)*vsRadius/8
		gz=math.floor(math.random()*8)*vsRadius/8
		x=gx y=vsYCurrent+vsBoostY z=gz
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="cluster" then
		x=(math.random()-0.5)*vsRadius*0.05
		y=vsYCurrent+vsBoostY
		z=(math.random()-0.5)*vsRadius*0.05
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="vortex" then
		vsAngle=(vsAngle+0.2)%(math.pi*2)
		r=vsRadius*(0.3+0.7*math.abs(math.sin(vsAngle*0.5)))
		x=(math.random()-0.5)*vsXDist+math.cos(vsAngle)*r
		y=vsYCurrent+vsBoostY
		z=(math.random()-0.5)*vsZDist+math.sin(vsAngle)*r
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="explosion" then
		dir=Vector3.new(math.random()-0.5,0,math.random()-0.5).Unit
		x=dir.X*vsXDist y=vsYCurrent+vsBoostY z=dir.Z*vsZDist
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="desync" then
		x=hrp.Position.X+(math.random()-0.5)*vsXDist
		y=vsYCurrent+vsBoostY
		z=hrp.Position.Z+(math.random()-0.5)*vsZDist
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="bounce" then
		vsStepDir=vsStepDir or 1
		vsYCurrent=vsYCurrent+vsYIncrement*vsStepDir
		if math.abs(vsYCurrent)>math.abs(vsYCustom)+vsYDist then vsStepDir=-vsStepDir end
		x=(math.random()-0.5)*2*vsXDist y=vsYCurrent+vsBoostY z=(math.random()-0.5)*2*vsZDist
	elseif pat=="wave" then
		vsAngle=(vsAngle+0.08)%(math.pi*2)
		x=vsXDist*math.sin(vsAngle)
		y=vsYCurrent+vsBoostY+vsRadius*0.1*math.sin(vsAngle*3)
		z=vsZDist*math.cos(vsAngle*0.5)
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="helix" then
		vsAngle=(vsAngle+0.1)%(math.pi*2)
		x=math.cos(vsAngle)*vsRadius
		y=vsYCurrent+vsBoostY+math.sin(vsAngle*0.5)*vsRadius*0.5
		z=math.sin(vsAngle)*vsRadius
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="star" then
		vsAngle=(vsAngle+0.13)%(math.pi*2)
		local sp=5
		local outer=vsRadius local inner=vsRadius*0.4
		local idx=math.floor(vsAngle/(math.pi*2/sp))
		local frac=(vsAngle%(math.pi*2/sp))/(math.pi*2/sp)
		local r1=frac<0.5 and outer or inner
		x=math.cos(vsAngle)*r1 y=vsYCurrent+vsBoostY z=math.sin(vsAngle)*r1
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="galaxy" then
		vsAngle=(vsAngle+0.07)%(math.pi*2)
		vsAngle2=(vsAngle2 or 0)+0.03
		local arm=math.floor(math.random()*3)*math.pi*2/3
		local r=vsRadius*(0.2+0.8*math.random())
		local twist=arm+r*0.002
		x=math.cos(twist)*r+(math.random()-0.5)*vsRadius*0.1
		y=vsYCurrent+vsBoostY+(math.random()-0.5)*vsRadius*0.02
		z=math.sin(twist)*r+(math.random()-0.5)*vsRadius*0.1
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="matrix" then
		local cols2=8 local rows=8
		local ci=math.floor(math.random()*cols2)
		local ri=math.floor(math.random()*rows)
		x=(ci-cols2/2)*vsRadius/cols2*2
		y=vsYCurrent+vsBoostY
		z=(ri-rows/2)*vsRadius/rows*2
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="strafe" then
		vsAngle=(vsAngle+0.15)%(math.pi*2)
		x=vsXDist*math.sin(vsAngle)
		y=vsYCurrent+vsBoostY
		z=0
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="cross" then
		vsAngle=(vsAngle+0.1)%(math.pi*2)
		local side=math.floor(vsAngle/(math.pi*0.5))%4
		local offsets2={{vsRadius,0},{0,vsRadius},{-vsRadius,0},{0,-vsRadius}}
		local o=offsets2[side+1]
		x=o[1] y=vsYCurrent+vsBoostY z=o[2]
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="diamond" then
		vsAngle=(vsAngle+0.1)%(math.pi*2)
		local t2=(vsAngle%(math.pi*0.5))/(math.pi*0.5)
		local side=math.floor(vsAngle/(math.pi*0.5))%4
		local corners2={{vsRadius,0},{0,vsRadius},{-vsRadius,0},{0,-vsRadius}}
		local c1=corners2[side+1] local c2=corners2[(side%4)+1]
		x=c1[1]+(c2[1]-c1[1])*t2 y=vsYCurrent+vsBoostY z=c1[2]+(c2[2]-c1[2])*t2
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="tornado" then
		vsAngle=(vsAngle+0.15)%(math.pi*2)
		local ri=(vsTeleportCount or 0)%20
		local r=vsRadius*(ri/20)
		x=math.cos(vsAngle)*r y=vsYCurrent+vsBoostY+ri*vsRadius*0.05 z=math.sin(vsAngle)*r
		vsTeleportCount=(vsTeleportCount or 0)+1
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="pendulum" then
		vsAngle=(vsAngle+0.12)%(math.pi*2)
		x=vsRadius*math.sin(vsAngle) y=vsYCurrent+vsBoostY z=0
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="random3d" then
		x=(math.random()-0.5)*2*vsXDist
		y=(math.random()-0.5)*2*vsYDist+vsBoostY
		z=(math.random()-0.5)*2*vsZDist
	elseif pat=="sphere" then
		local theta=math.random()*math.pi*2
		local phi=math.acos(2*math.random()-1)
		x=vsRadius*math.sin(phi)*math.cos(theta)
		y=vsYCurrent+vsBoostY+vsRadius*math.cos(phi)
		z=vsRadius*math.sin(phi)*math.sin(theta)
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="cube" then
		local face=math.random(1,6)
		local u=(math.random()-0.5)*2*vsRadius local v2=(math.random()-0.5)*2*vsRadius
		if face==1 then x=vsRadius y=vsYCurrent+vsBoostY+u z=v2
		elseif face==2 then x=-vsRadius y=vsYCurrent+vsBoostY+u z=v2
		elseif face==3 then x=u y=vsYCurrent+vsBoostY+vsRadius z=v2
		elseif face==4 then x=u y=vsYCurrent+vsBoostY-vsRadius z=v2
		elseif face==5 then x=u y=vsYCurrent+vsBoostY+v2 z=vsRadius
		else x=u y=vsYCurrent+vsBoostY+v2 z=-vsRadius end
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	elseif pat=="noise" then
		local t3=tick()
		x=math.sin(t3*1.3)*vsXDist+(math.random()-0.5)*vsXDist*0.5
		y=vsYCurrent+vsBoostY+math.cos(t3*0.7)*vsRadius*0.1
		z=math.cos(t3*1.7)*vsZDist+(math.random()-0.5)*vsZDist*0.5
		if vsYIncrEnabled then vsYCurrent=vsYCurrent+vsYIncrement end
	else
		x=(math.random()-0.5)*2*vsXDist
		y=(math.random()-0.5)*2*vsYDist
		z=(math.random()-0.5)*2*vsZDist
	end
	return applyVSModifiers(x,y,z)
end

-- core tp function for orbit
function computeORBCoords(hrp,epos)
	x,y,z=0,0,0
	pat=orbPattern
	local bx=orbRelativeToTarget and epos.X or 0
	local by=orbRelativeToTarget and epos.Y or 0
	local bz=orbRelativeToTarget and epos.Z or 0
	if pat=="custom" then
		x=orbCustomX y=orbCustomY z=orbCustomZ
		return x,y,z
	elseif pat=="static" then
		x=orbStaticX y=orbStaticY+orbBoostY z=orbStaticZ
	elseif pat=="void" then
		x=bx+(math.random()-0.5)*2*orbXDist
		y=orbYCurrent+orbBoostY
		z=bz+(math.random()-0.5)*2*orbZDist
		if orbYIncrEnabled then orbYCurrent=orbYCurrent+orbYIncrement end
	elseif pat=="everywhere" then
		x=(math.random()-0.5)*2*orbXDist
		y=(math.random()-0.5)*2*orbYDist
		z=(math.random()-0.5)*2*orbZDist
	elseif pat=="random" then
		x=bx+(math.random()-0.5)*orbRadius*2
		y=by+orbHeight+(math.random()-0.5)*orbRadius*0.5
		z=bz+(math.random()-0.5)*orbRadius*2
	elseif pat=="circular" or pat=="target" then
		orbAngle=(orbAngle+0.15)%(math.pi*2)
		x=bx+math.cos(orbAngle)*orbRadius
		y=by+orbHeight
		z=bz+math.sin(orbAngle)*orbRadius
	elseif pat=="figure8" then
		orbAngle=(orbAngle+0.1)%(math.pi*2)
		x=bx+orbRadius*math.sin(orbAngle)
		y=by+orbHeight
		z=bz+orbRadius*math.sin(orbAngle)*math.cos(orbAngle)
	elseif pat=="spiral" then
		orbAngle=(orbAngle+0.12)%(math.pi*2)
		r=orbRadius*(0.3+0.7*math.abs(math.sin(orbAngle*0.4)))
		x=bx+math.cos(orbAngle)*r
		y=by+orbHeight+math.sin(orbAngle)*3
		z=bz+math.sin(orbAngle)*r
	elseif pat=="zigzag" then
		orbAngle=(orbAngle+0.12)%(math.pi*2)
		x=bx+orbRadius*math.sin(orbAngle)
		y=by+orbHeight
		z=bz+orbRadius*math.sin(orbAngle*3)*0.3
	elseif pat=="helix" then
		orbAngle=(orbAngle+0.1)%(math.pi*2)
		x=bx+math.cos(orbAngle)*orbRadius
		y=by+orbHeight+math.sin(orbAngle*0.5)*orbRadius*0.5
		z=bz+math.sin(orbAngle)*orbRadius
	elseif pat=="square" then
		orbAngle=(orbAngle+0.05)%(math.pi*2)
		side=math.floor(orbAngle/(math.pi*0.5))%4
		t2=(orbAngle%(math.pi*0.5))/(math.pi*0.5)
		corners={{1,1},{-1,1},{-1,-1},{1,-1}}
		c1=corners[side+1] c2=corners[(side%4)+1]
		x=bx+(c1[1]+(c2[1]-c1[1])*t2)*orbRadius
		y=by+orbHeight
		z=bz+(c1[2]+(c2[2]-c1[2])*t2)*orbRadius
	elseif pat=="triangle" then
		orbAngle=(orbAngle+0.07)%(math.pi*2)
		side=math.floor(orbAngle/(math.pi*2/3))%3
		t2=(orbAngle%(math.pi*2/3))/(math.pi*2/3)
		corners={}
		for i=0,2 do a=math.pi*2/3*i table.insert(corners,{math.cos(a)*orbRadius,math.sin(a)*orbRadius}) end
		c1=corners[side+1] c2=corners[(side%3)+1]
		x=bx+(c1[1]+(c2[1]-c1[1])*t2)
		y=by+orbHeight
		z=bz+(c1[2]+(c2[2]-c1[2])*t2)
	elseif pat=="vortex" then
		orbAngle=(orbAngle+0.2)%(math.pi*2)
		r=orbRadius*(0.3+0.7*math.abs(math.sin(orbAngle*0.5)))
		x=(math.random()-0.5)*orbXDist+math.cos(orbAngle)*r
		y=orbYCurrent+orbBoostY
		z=(math.random()-0.5)*orbZDist+math.sin(orbAngle)*r
		if orbYIncrEnabled then orbYCurrent=orbYCurrent+orbYIncrement end
	elseif pat=="desync" then
		x=bx+(math.random()-0.5)*orbXDist
		y=orbYCurrent+orbBoostY
		z=bz+(math.random()-0.5)*orbZDist
		if orbYIncrEnabled then orbYCurrent=orbYCurrent+orbYIncrement end
	elseif pat=="bounce" then
		orbStepDir=orbStepDir or 1
		orbYCurrent=orbYCurrent+orbYIncrement*orbStepDir
		if math.abs(orbYCurrent)>math.abs(orbYCustom)+orbYDist then orbStepDir=-orbStepDir end
		x=bx+(math.random()-0.5)*orbRadius*2 y=orbYCurrent+orbBoostY z=bz+(math.random()-0.5)*orbRadius*2
	elseif pat=="wave" then
		orbAngle=(orbAngle+0.08)%(math.pi*2)
		x=bx+orbRadius*math.sin(orbAngle)
		y=by+orbHeight+orbRadius*0.3*math.sin(orbAngle*2)
		z=bz+orbRadius*math.cos(orbAngle*0.5)
	elseif pat=="cross" then
		orbAngle=(orbAngle+0.1)%(math.pi*2)
		local side2=math.floor(orbAngle/(math.pi*0.5))%4
		local offsets3={{orbRadius,0},{0,orbRadius},{-orbRadius,0},{0,-orbRadius}}
		local o2=offsets3[side2+1]
		x=bx+o2[1] y=by+orbHeight z=bz+o2[2]
	elseif pat=="diamond" then
		orbAngle=(orbAngle+0.1)%(math.pi*2)
		local t3=(orbAngle%(math.pi*0.5))/(math.pi*0.5)
		local side3=math.floor(orbAngle/(math.pi*0.5))%4
		local corners3={{orbRadius,0},{0,orbRadius},{-orbRadius,0},{0,-orbRadius}}
		local c3=corners3[side3+1] local c4=corners3[(side3%4)+1]
		x=bx+c3[1]+(c4[1]-c3[1])*t3 y=by+orbHeight z=bz+c3[2]+(c4[2]-c3[2])*t3
	elseif pat=="tornado" then
		orbAngle=(orbAngle+0.15)%(math.pi*2)
		local ri2=(vsTeleportCount or 0)%20
		local r2=orbRadius*(ri2/20)
		x=bx+math.cos(orbAngle)*r2 y=by+orbHeight+ri2*orbRadius*0.05 z=bz+math.sin(orbAngle)*r2
		vsTeleportCount=(vsTeleportCount or 0)+1
	elseif pat=="pendulum" then
		orbAngle=(orbAngle+0.12)%(math.pi*2)
		x=bx+orbRadius*math.sin(orbAngle) y=by+orbHeight z=bz
	elseif pat=="random3d" then
		x=(math.random()-0.5)*2*orbXDist
		y=(math.random()-0.5)*2*orbYDist
		z=(math.random()-0.5)*2*orbZDist
	elseif pat=="sphere" then
		local theta=math.random()*math.pi*2
		local phi=math.acos(2*math.random()-1)
		x=bx+orbRadius*math.sin(phi)*math.cos(theta)
		y=by+orbHeight+orbRadius*math.cos(phi)
		z=bz+orbRadius*math.sin(phi)*math.sin(theta)
	elseif pat=="cube" then
		local face=math.random(1,6)
		local u=(math.random()-0.5)*2*orbRadius local v2=(math.random()-0.5)*2*orbRadius
		if face==1 then x=bx+orbRadius y=by+orbHeight+u z=bz+v2
		elseif face==2 then x=bx-orbRadius y=by+orbHeight+u z=bz+v2
		elseif face==3 then x=bx+u y=by+orbHeight+orbRadius z=bz+v2
		elseif face==4 then x=bx+u y=by+orbHeight-orbRadius z=bz+v2
		elseif face==5 then x=bx+u y=by+orbHeight+v2 z=bz+orbRadius
		else x=bx+u y=by+orbHeight+v2 z=bz-orbRadius end
	elseif pat=="noise" then
		local t4=tick()
		x=bx+math.sin(t4*1.3)*orbRadius+(math.random()-0.5)*orbRadius*0.3
		y=by+orbHeight+math.cos(t4*0.7)*orbRadius*0.3
		z=bz+math.cos(t4*1.7)*orbRadius+(math.random()-0.5)*orbRadius*0.3
	elseif pat=="star" then
		orbAngle=(orbAngle+0.13)%(math.pi*2)
		local sp=5
		local outer=orbRadius local inner=orbRadius*0.4
		local frac2=(orbAngle%(math.pi*2/sp))/(math.pi*2/sp)
		local r3=frac2<0.5 and outer or inner
		x=bx+math.cos(orbAngle)*r3 y=by+orbHeight z=bz+math.sin(orbAngle)*r3
	elseif pat=="galaxy" then
		local arm2=math.floor(math.random()*3)*math.pi*2/3
		local r4=orbRadius*(0.2+0.8*math.random())
		local twist2=arm2+r4*0.002
		x=bx+math.cos(twist2)*r4+(math.random()-0.5)*orbRadius*0.1
		y=by+orbHeight+(math.random()-0.5)*orbRadius*0.05
		z=bz+math.sin(twist2)*r4+(math.random()-0.5)*orbRadius*0.1
	elseif pat=="matrix" then
		local cols3=8 local rows3=8
		local ci3=math.floor(math.random()*cols3)
		local ri3=math.floor(math.random()*rows3)
		x=bx+(ci3-cols3/2)*orbRadius/cols3*2
		y=by+orbHeight
		z=bz+(ri3-rows3/2)*orbRadius/rows3*2
	elseif pat=="strafe" then
		orbAngle=(orbAngle+0.15)%(math.pi*2)
		x=bx+orbRadius*math.sin(orbAngle) y=by+orbHeight z=bz
	elseif pat=="ellipse" then
		orbAngle=(orbAngle+0.1)%(math.pi*2)
		x=bx+orbRadius*math.cos(orbAngle)
		y=by+orbHeight
		z=bz+orbRadius*0.5*math.sin(orbAngle)
	else
		x=(math.random()-0.5)*2*orbXDist
		y=(math.random()-0.5)*2*orbYDist
		z=(math.random()-0.5)*2*orbZDist
	end
	return applyORBModifiers(x,y,z)
end

-- HEARTBEAT
rs.Heartbeat:Connect(function(dt)
	chr=lp.Character if not chr then return end
	hrp=chr:FindFirstChild("HumanoidRootPart") if not hrp then return end

	if (S.void or S.orbit) and vsNoclip then
		pcall(function()
			for _,p in ipairs(chr:GetDescendants()) do
				if p:IsA("BasePart") then p.CanCollide=false end
			end
		end)
	end

	-- VOID SPAM
	if S.void then
		if not savedNormalPos then savedNormalPos=hrp.CFrame end
		tvs=tvs+dt
		if tvs>=vsSpeed then
			tvs=0
			pcall(function()
				ox,oy,oz=computeVSCoords(hrp)
				if not vsFreeze then
					hrp.CFrame=CFrame.new(ox,oy,oz)
					hrp.AssemblyLinearVelocity=Vector3.zero
					if vsMultiTP then
						for mi=1,vsMultiCount-1 do
							task.delay(vsMultiDelay*mi,function()
								mx,my,mz=computeVSCoords(hrp)
								pcall(function()
									hrp.CFrame=CFrame.new(mx,my,mz)
									hrp.AssemblyLinearVelocity=Vector3.zero
								end)
							end)
						end
					end
				end
			end)
		end
	else
		savedNormalPos=nil tvs=0
	end

	-- ORBIT
	if S.orbit then
		if not savedOrbitPos then savedOrbitPos=hrp.CFrame end
		torb=torb+dt
		if torb>=orbSpeed then
			torb=0
			pcall(function()
				epos=Vector3.new(0,0,0)
				if orbTarget=="nearest" then
					ne,_=getNearest()
					if ne then nhrp2=ne:FindFirstChild("HumanoidRootPart") if nhrp2 then epos=nhrp2.Position end end
				elseif orbTarget=="random" then
					pls=game:GetService("Players"):GetPlayers() valid={}
					for _,pl in ipairs(pls) do
						if pl~=lp and pl.Character then h=pl.Character:FindFirstChild("HumanoidRootPart") if h then table.insert(valid,h.Position) end end
					end
					if #valid>0 then epos=valid[math.random(1,#valid)] end
				elseif orbTarget=="self" then
					epos=hrp.Position
				elseif orbTarget=="furthest" then
					fbd=0
					for _,pl in ipairs(game:GetService("Players"):GetPlayers()) do
						if pl~=lp and pl.Character then
							fh=pl.Character:FindFirstChild("HumanoidRootPart")
							if fh then
								fd=(fh.Position-hrp.Position).Magnitude
								if fd>fbd then fbd=fd epos=fh.Position end
							end
						end
					end
				end
				ox,oy,oz=computeORBCoords(hrp,epos)
				hrp.CFrame=CFrame.new(ox,oy,oz)
				hrp.AssemblyLinearVelocity=Vector3.zero
				if orbMultiTP then
					for mi=1,orbMultiCount-1 do
						task.delay(orbMultiDelay*mi,function()
							mx2,my2,mz2=computeORBCoords(hrp,epos)
							pcall(function()
								hrp.CFrame=CFrame.new(mx2,my2,mz2)
								hrp.AssemblyLinearVelocity=Vector3.zero
							end)
						end)
					end
				end
			end)
		end
	else
		savedOrbitPos=nil torb=0
	end

	-- TP AURA
	if S.tpa then
		ttpa=ttpa+dt
		if ttpa>=(tpaSpeed2 or 0.05) then ttpa=0
			best,bd=nil,math.huge
			for _,pl in ipairs(game:GetService("Players"):GetPlayers()) do
				if pl~=lp and pl.Character then
					h=pl.Character:FindFirstChild("HumanoidRootPart")
					hum=pl.Character:FindFirstChildOfClass("Humanoid")
					if h and hum and hum.Health>0 then
						d=(h.Position-hrp.Position).Magnitude
						if d<bd then best=pl.Character bd=d end
					end
				end
			end
			if best then
				head2=best:FindFirstChild("Head") or best:FindFirstChild("HumanoidRootPart")
				if head2 then
					tpaAng=(tpaAng+15)%360
					len=_G.tpaLen or tpaLen
					pcall(function()
						hrp.CFrame=CFrame.new(
							head2.Position.X+math.cos(math.rad(tpaAng))*len,
							head2.Position.Y+len,
							head2.Position.Z+math.sin(math.rad(tpaAng))*len
						)
					end)
				end
			end
		end
	end

	-- MOTION
	if S.motion then
		motionT=motionT+dt
		re2=getAny()
		epos=re2 and re2:FindFirstChild("HumanoidRootPart") and re2:FindFirstChild("HumanoidRootPart").Position or hrp.Position
		pcall(function()
			if motionPattern=="orbit" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				hrp.CFrame=CFrame.new(epos.X+math.cos(motionAng)*motionR,epos.Y+5,epos.Z+math.sin(motionAng)*motionR)
			elseif motionPattern=="figure8" then
				motionAng=(motionAng+motionSpeed*0.5*dt)%(math.pi*2)
				hrp.CFrame=CFrame.new(epos.X+math.sin(motionAng)*motionR,epos.Y+3,epos.Z+math.sin(motionAng*2)*motionR*0.5)
			elseif motionPattern=="spiral" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				r=motionR*(0.5+0.5*math.abs(math.sin(motionT*0.5)))
				hrp.CFrame=CFrame.new(epos.X+math.cos(motionAng)*r,epos.Y+motionT*0.3,epos.Z+math.sin(motionAng)*r)
				if motionT>10 then motionT=0 end
			elseif motionPattern=="bounce" then
				hrp.CFrame=CFrame.new(hrp.Position.X,hrp.Position.Y+(motionSpeed*dt*motionBounceDir),hrp.Position.Z)
				if hrp.Position.Y>epos.Y+motionR then motionBounceDir=-1 elseif hrp.Position.Y<epos.Y-motionR then motionBounceDir=1 end
			elseif motionPattern=="random" then
				if motionT>=0.1 then motionT=0 hrp.CFrame=CFrame.new(epos.X+(math.random()-0.5)*motionR*2,epos.Y+math.random(0,8),epos.Z+(math.random()-0.5)*motionR*2) end
			elseif motionPattern=="desync" then
				if motionT>=0.04 then motionT=0 hrp.CFrame=CFrame.new(hrp.Position.X+(math.random()-0.5)*motionR,hrp.Position.Y,hrp.Position.Z+(math.random()-0.5)*motionR) end
			elseif motionPattern=="cross" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				side=math.floor(motionAng/(math.pi*0.5))%4
				offsets={{1,0},{0,1},{-1,0},{0,-1}} o=offsets[side+1]
				hrp.CFrame=CFrame.new(epos.X+o[1]*motionR,epos.Y+3,epos.Z+o[2]*motionR)
			elseif motionPattern=="tpspam" then
				hrp.CFrame=CFrame.new(epos.X+(math.random()-0.5)*motionR*2,epos.Y+(math.random()-0.5)*4,epos.Z+(math.random()-0.5)*motionR*2)
			elseif motionPattern=="wave" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				hrp.CFrame=CFrame.new(epos.X+motionR*math.sin(motionAng),epos.Y+motionR*0.5*math.sin(motionAng*2),epos.Z+motionR*math.cos(motionAng*0.5))
			elseif motionPattern=="helix" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				hrp.CFrame=CFrame.new(epos.X+math.cos(motionAng)*motionR,epos.Y+motionT*0.5,epos.Z+math.sin(motionAng)*motionR)
				if motionT>8 then motionT=0 end
			elseif motionPattern=="square" then
				motionAng=(motionAng+motionSpeed*0.3*dt)%(math.pi*2)
				side=math.floor(motionAng/(math.pi*0.5))%4
				offsets={{1,0},{0,1},{-1,0},{0,-1}} o=offsets[side+1]
				hrp.CFrame=CFrame.new(epos.X+o[1]*motionR,epos.Y+3,epos.Z+o[2]*motionR)
			elseif motionPattern=="pendulum" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				hrp.CFrame=CFrame.new(epos.X+motionR*math.sin(motionAng),epos.Y+5,epos.Z)
			elseif motionPattern=="tornado" then
				motionAng=(motionAng+motionSpeed*dt)%(math.pi*2)
				r=motionR*(0.2+0.8*(motionT%5)/5)
				hrp.CFrame=CFrame.new(epos.X+math.cos(motionAng)*r,epos.Y+3+motionT*0.4,epos.Z+math.sin(motionAng)*r)
				if motionT>6 then motionT=0 end
			end
		end)
	end

	-- ANTI AIM
	if S.antiaim then
		pcall(function()
			if aaMode=="spinbot" then aaAng=(aaAng+25)%360 hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(aaAng),0)
			elseif aaMode=="fast spin" then aaAng=(aaAng+90)%360 hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(aaAng),0)
			elseif aaMode=="jitter" then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(math.random(0,360)),0)
			elseif aaMode=="backwards" then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(180),0)
			elseif aaMode=="sideways" then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(90),0)
			elseif aaMode=="pitch up" then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(89),0,0)
			elseif aaMode=="pitch down" then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(-89),0,0)
			elseif aaMode=="fake lag" then if math.random(1,3)==1 then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(math.random(0,360)),0) end
			elseif aaMode=="360 jitter" then aaAng=(aaAng+45)%360 hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(math.random(-30,30)),math.rad(aaAng),math.rad(math.random(-15,15)))
			elseif aaMode=="matrix" then hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(math.random(-180,180)),math.rad(math.random(-180,180)),math.rad(math.random(-180,180)))
			elseif aaMode=="random" then aaAng=math.random(0,360) hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(0,math.rad(aaAng),0)
			elseif aaMode=="wobble" then aaAng=(aaAng+15)%360 hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(math.sin(aaAng/30)*20),math.rad(aaAng),0)
			elseif aaMode=="drunk" then aaAng=(aaAng+8)%360 hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(math.sin(aaAng/20)*15),math.rad(aaAng*2),math.rad(math.cos(aaAng/25)*10))
			elseif aaMode=="flip" then aaAng=(aaAng+60)%360 flip=math.floor(aaAng/60)%2==0 and 0 or 180 hrp.CFrame=CFrame.new(hrp.Position)*CFrame.Angles(math.rad(flip),math.rad(aaAng),0)
			end
		end)
	end

	if S.bypass and not bpLock then
		taf=taf+dt
		if taf>=0.04+math.random()*0.01 then taf=0
			pcall(function()
				for _,t in ipairs(chr:GetChildren()) do
					if t:IsA("Tool") then t:Activate() if rfDouble then t:Activate() end end
				end
			end)
		end
	end

	if S.riotAbuse and riotSpam then
		pcall(function()
			e=getAny()
			if e then
				shield=riotCache[e]
				if shield and shield.Parent then
					myhrp=chr:FindFirstChild("HumanoidRootPart")
					if myhrp then firetouchinterest(myhrp,shield,0) firetouchinterest(myhrp,shield,1) end
				end
			end
		end)
	end

	if S.rivalsBypass then
		pcall(function()
			e,ep=getNearest() if e then doRivalsBypass(e) end
		end)
	end
end)

function doHit(obj)
	pcall(function()
		if not obj or not obj.Parent then return end
		e=getAny() if not e then return end
		ehrp=e:FindFirstChild("HumanoidRootPart") if not ehrp then return end
		torso=e:FindFirstChild("UpperTorso") or e:FindFirstChild("Torso") or ehrp
		head=e:FindFirstChild("Head")
		chr2=lp.Character if not chr2 then return end
		myhrp=chr2:FindFirstChild("HumanoidRootPart") if not myhrp then return end
		vel=Vector3.zero pcall(function()vel=ehrp.AssemblyLinearVelocity end)
		tpos=torso.Position+vel*0.05
		pcall(function() obj.CFrame=CFrame.new(tpos) obj.Velocity=Vector3.zero obj.CanCollide=false end)
		function hitPart(p)
			if not p then return end
			if bpMode=="firetouchinterest" or bpMode=="both" or bpMode=="multi" or bpMode=="spam" then
				pcall(firetouchinterest,obj,p,0) pcall(firetouchinterest,obj,p,1)
				pcall(firetouchinterest,myhrp,p,0) pcall(firetouchinterest,myhrp,p,1)
				if bpMode=="spam" then for _=1,5 do pcall(firetouchinterest,obj,p,0) pcall(firetouchinterest,obj,p,1) end end
			end
			if bpMode=="getconnections" or bpMode=="both" or bpMode=="multi" then
				pcall(function() for _,c in ipairs(getconnections(p.Touched)) do pcall(c.Fire,c,obj) end end)
			end
			if bpMode=="velocity" then
				pcall(function() dir=(p.Position-obj.Position).Unit obj.AssemblyLinearVelocity=dir*500 end)
				pcall(firetouchinterest,obj,p,0) pcall(firetouchinterest,obj,p,1)
			end
			if bpMultiHit then
				task.delay(bpHitDelay,function() pcall(firetouchinterest,obj,p,0) pcall(firetouchinterest,obj,p,1) end)
				task.delay(bpHitDelay*2,function() pcall(firetouchinterest,obj,p,0) pcall(firetouchinterest,obj,p,1) end)
			end
		end
		hitPart(torso) hitPart(ehrp) if head then hitPart(head) end
		if bpMode=="multi" then
			for _,p in ipairs(e:GetDescendants()) do
				if p:IsA("BasePart") then pcall(firetouchinterest,obj,p,0) pcall(firetouchinterest,obj,p,1) end
			end
		end
		if S.riotAbuse then
			pcall(function()
				shield=riotCache[e]
				if shield and shield.Parent then
					if riotMode=="touched" then hitPart(shield) for _,c in ipairs(getconnections(shield.Touched)) do pcall(c.Fire,c,obj) end
					elseif riotMode=="destroy" then shield.Size=Vector3.new(0.01,0.01,0.01) shield.CFrame=CFrame.new(tpos) hitPart(shield)
					elseif riotMode=="both" then hitPart(shield) shield.CFrame=CFrame.new(tpos) hitPart(shield) for _,c in ipairs(getconnections(shield.Touched)) do pcall(c.Fire,c,obj) end
					elseif riotMode=="spam" then for _=1,10 do pcall(firetouchinterest,obj,shield,0) pcall(firetouchinterest,obj,shield,1) pcall(firetouchinterest,myhrp,shield,0) pcall(firetouchinterest,myhrp,shield,1) end
					elseif riotMode=="detach" then pcall(function()shield.Parent=workspace end) hitPart(shield)
					elseif riotMode=="velocity" then pcall(function() dir=(tpos-shield.Position).Unit shield.AssemblyLinearVelocity=dir*1000 end) hitPart(shield)
					end
				end
			end)
		end
	end)
end

function hookBP()
	workspace.DescendantAdded:Connect(function(obj)
		if not S.bypass then return end
		if obj.Name~="SDSDSD" and obj.Name~="Sound" then return end
		if snapped[obj] then return end
		snapped[obj]=true
		doHit(obj) task.defer(function()doHit(obj)end)
		task.delay(bpHitDelay,function()doHit(obj)end)
		task.delay(bpHitDelay*2,function()doHit(obj)end)
	end)
end

function hookTools(chr2)
	function hook(tool)
		if not tool:IsA("Tool") then return end
		tool.Activated:Connect(function()
			if not S.bypass then return end
			myhrp2=chr2:FindFirstChild("HumanoidRootPart") if not myhrp2 then return end
			re=getAny() if not re then return end
			rehrp2=re:FindFirstChild("HumanoidRootPart") if not rehrp2 then return end
			rtorso2=re:FindFirstChild("UpperTorso") or re:FindFirstChild("Torso") or rehrp2
			preSavedCF=myhrp2.CFrame bpLock=true
			pcall(function() myhrp2.CFrame=CFrame.new(rtorso2.Position)*CFrame.new(0,0,bpDistance*0.1) end)
		end)
	end
	for _,t in ipairs(chr2:GetChildren()) do hook(t) end
	chr2.ChildAdded:Connect(hook)
end
